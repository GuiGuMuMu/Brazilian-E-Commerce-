# ## Optional: logistic regression model
#
# This is the "not too simple statistics" section.
# We use logistic regression to understand low-review risk.

# %%
# pip install statsmodels

import statsmodels.formula.api as smf

model_df = order_fact[
    (order_fact["order_status"] == "delivered")
    & order_fact["delivery_days_clean"].notna()
    & order_fact["payment_value"].notna()
    & order_fact["freight_value"].notna()
].copy()

model = smf.logit(
    formula="low_review ~ delivery_days_clean + payment_value + freight_value",
    data=model_df
).fit()

print(model.summary())

# %%
odds_ratios = np.exp(model.params)
odds_ratios
